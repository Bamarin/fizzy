def expand_variables
  vars_regexp = /\$([a-zA-Z_]+[a-zA-Z0-9_]*)|\$\{(.+)\}/
  self.gsub(vars_regexp) { ENV[$1||$2] }
end

def titleize!
  self.replace(self.split(/[_-]/).each { |s| s.capitalize! }.join(""))
end
