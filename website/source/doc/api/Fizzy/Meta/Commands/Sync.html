def validate!(spec)
  # 1: Validate.
  invalid_spec :repo unless spec.has_key?(:repo)
  invalid_spec :dst  unless spec.has_key?(:dst)

  # 2: Normalize.
  @repo = spec[:repo]
  @dst = Pathname.new(spec[:dst]).expand_variables.expand_path
end
