def parse(value, type: nil)
  case @stack.pop
  when :list
    if value.is_a? Array
      value.map { |e| parse(e ) }
    else
      error "Invalid untyped value `#{value}`: cannot parse into a list"
    end
  else typize(value)
  end
end
